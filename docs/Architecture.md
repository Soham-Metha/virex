VIREX

├── install.sh                            # Install script for linux
├── LICENSE                               # GPL-3
├── Makefile                              # RULES: virex, sasm, occ, html, clean
├── mkdocs.yml

    ├── fonts/

    └── themes/

        └── vs_code/
